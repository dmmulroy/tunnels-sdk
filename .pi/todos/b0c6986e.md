## Goal

Create the production and test layer compositions, and the `effect/index.ts` barrel file that defines the public API surface of `tunnel-sdk/effect`.

## Context

All services are implemented at this point. This todo wires them together and defines what consumers import.

## What to create

### File: `packages/tunnel-sdk/src/effect/layers/Live.ts`

```ts
import { Layer } from "effect"
import { FetchHttpClient } from "effect/unstable/http"
import { NodeServices } from "@effect/platform-node"
import { CloudflareApi, CloudflareApiConfig } from "../services/CloudflareApi.js"
import { TunnelOperations } from "../services/TunnelOperations.js"
import { IngressManager } from "../services/IngressManager.js"
import { DnsManager } from "../services/DnsManager.js"
import { RouteManager } from "../services/RouteManager.js"
import { VNetManager } from "../services/VNetManager.js"
import { TunnelProcessService } from "../services/TunnelProcess.js"
import { CloudflaredBinary } from "../services/CloudflaredBinary.js"

// Explicit config — all services wired for production
export const LiveLayer = (config: CloudflareApiConfig) => Layer.mergeAll(
  TunnelOperations.layer,
  IngressManager.layer,
  DnsManager.layer,
  RouteManager.layer,
  VNetManager.layer,
  TunnelProcessService.layer,
  CloudflaredBinary.layer,
).pipe(
  Layer.provide(CloudflareApi.layer(config)),
  Layer.provide(FetchHttpClient.layer),
  Layer.provide(NodeServices.layer),
)

// From environment — reads CF_ACCOUNT_ID, CF_API_TOKEN, CF_BASE_URL
export const LiveLayerFromEnv = Layer.mergeAll(
  TunnelOperations.layer,
  IngressManager.layer,
  DnsManager.layer,
  RouteManager.layer,
  VNetManager.layer,
  TunnelProcessService.layer,
  CloudflaredBinary.layer,
).pipe(
  Layer.provide(CloudflareApi.layerFromEnv),
  Layer.provide(FetchHttpClient.layer),
  Layer.provide(NodeServices.layer),
)
```

Verify the resulting Layer type satisfies:
```ts
Layer.Layer<
  TunnelOperations | IngressManager | DnsManager | RouteManager |
  VNetManager | TunnelProcessService | CloudflaredBinary,
  never  // (or Config.ConfigError for LiveLayerFromEnv)
>
```

### Test: `packages/tunnel-sdk/src/effect/index.test.ts`

A smoke test that verifies the layer composition works end-to-end:

```ts
import { assert, it } from "@effect/vitest"
import { Effect } from "effect"
import { TunnelOperations, TestLayer } from "./index.js"

it.effect("TestLayer provides all services", () =>
  Effect.gen(function*() {
    // Just verify we can yield the service — it proves the layer wiring
    const ops = yield* TunnelOperations
    assert.isDefined(ops)
  }).pipe(Effect.provide(TestLayer))
)
```

Also verify that `LiveLayer` type-checks (even if we can't run it without real credentials):
```ts
import { LiveLayer, CloudflareApiConfig } from "./index.js"
import { Redacted } from "effect"

// This just needs to compile — proves layer composition is sound
const _layer = LiveLayer(new CloudflareApiConfig({
  accountId: "test",
  apiToken: Redacted.make("test-token"),
}))
```
